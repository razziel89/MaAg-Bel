53  class OBAPI OBSqrtTbl
54  {
55  double _max,_incr,*_tbl;
56  public:
57  OBSqrtTbl():
58  _max(0.0), _incr(0.0), _tbl(NULL)
59  { }
64  OBSqrtTbl(const double max, const double incr):
65  _max(max*max), _incr(incr), _tbl(NULL)
66  {
67  Init(max,incr);
68  }
69  ~OBSqrtTbl()
70  {
71  if (_tbl)
72  {
73  delete [] _tbl;
74  _tbl = NULL;
75  }
76  }
79  double Sqrt(double d2) const
80  {
81  if (_tbl)
82  return((d2 < _max) ? _tbl[static_cast<int>(d2*_incr)]:sqrt(d2));
83  else
84  return 0.0;
85  }
89  void Init(double max,double incr)
90  {
91  // parameters are potentially unneeded, but let's do this until we can
92  // deprecate them
93  _max = max * max;
94  _incr = incr;
95 
96  //array size needs to be large enough to account for fp error
97  int i;
98  double r;
99  _tbl = new double [static_cast<int>((_max/_incr)+10)];
100  for (r = (_incr/2.0),i=0;r <= _max;r += _incr,++i)
101  _tbl[i] = sqrt(r);
102 
103  _incr = 1/_incr;
104  }
105  };
